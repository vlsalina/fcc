function chunkArrayInGroups(arr, size) { var ra = []; var temp = []; var len = arr.length; var counter = 1; for (var i = 0; i < len; i++) { if (counter > size) { counter = 1; ra.push(temp); temp = []; } temp.push(arr[i]); counter++; if (i == (len-1)) { ra.push(temp); break; } } return ra; } chunkArrayInGroups(["a", "b", "c", "d"], 2);
